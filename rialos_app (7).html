// RialOS Payroll Engine — Rialo Integration import { RialOS } from '@rialos/sdk' // Initialize with your Rialo account const flow = new RialOS({ rialoid: 'your-rialo-user-id', network: 'devnet' }) // Bulk disburse — instant settlement const result = await flow.payroll.disburse({ employees: [ { id: 'emp_001', amount: 5000, currency: 'USDC' }, { id: 'emp_002', amount: 7200, currency: 'USDC' }, ], stream: true, // per-second streaming via Rialo memo: 'April 2026 Payroll' }) console.log(result.txHash, result.settled) // → { txHash: '0xabc...', settled: true, instant: true } // Stream salary in real-time const stream = await flow.payroll.stream({ recipient: '0xEmployee...', ratePerSecond: 0.001929, // $5000/month currency: 'USDC' }) stream.on('tick', ({ sent, elapsed }) => { console.log(`Streamed: $${sent} USDC`) })
